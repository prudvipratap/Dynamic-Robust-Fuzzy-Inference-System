System Architecture – 4-Layer Hierarchical DRFIS

The Dynamic Robust Fuzzy Inference System addresses key gaps in conventional FIS: no integrated uncertainty handling, no adaptive robustness analysis, and fixed rule sets. DRFIS introduces 4 modular layers with self-calibrating rules, real-time uncertainty propagation, and stability-guaranteed inference.

Layer 1 – Input Processing Layer

Standardizes raw data, detects uncertainty, applies variance analysis and adaptive transformations.

##############################################################
#  DRFIS – Layer 1: Input Processing Layer                  #
#  Author : Kanakala Prudvi Pratap, LPU                     #
##############################################################

import numpy as np
from dataclasses import dataclass, field
from typing import List, Dict, Tuple, Optional

# ─── Data Structures ────────────────────────────────────────

@dataclass
class InputBundle:
    """Holds a single input sample plus its uncertainty metadata."""
    raw_values  : np.ndarray
    normalized  : np.ndarray = field(default_factory=lambda: np.array([]))
    aleatory_u  : float = 0.0   # Ua – statistical variance
    epistemic_u : float = 0.0   # Ue – systematic uncertainty
    total_u     : float = 0.0   # Utotal = Ua + Ue  (Eq. 8)
    delta       : float = 0.0   # δ – uncertainty parameter (Eq. 5)

# ─── Input Processing Layer ─────────────────────────────────

class InputProcessingLayer:
    """
    First layer of DRFIS.
    Responsibilities:
      1. Normalize / feature-scale inputs
      2. Compute aleatory uncertainty  Ua  (Eq. 9)
      3. Compute epistemic uncertainty Ue  (Eq. 10)
      4. Compute total uncertainty     Utotal (Eq. 8)
      5. Compute uncertainty parameter δ   (Eq. 5)
    """

    def __init__(self, feature_range: Tuple[float,float] = (0.0, 1.0)):
        self.feature_range = feature_range
        self.scaler_min: Optional[np.ndarray] = None
        self.scaler_max: Optional[np.ndarray] = None
        self.sigma_inherent : float = 0.1   # σi  – inherent noise std
        self.sigma_env      : float = 0.05  # σe  – environmental noise

    def fit(self, dataset: np.ndarray) -> "InputProcessingLayer":
        """Compute normalization bounds from a training dataset."""
        self.scaler_min = dataset.min(axis=0)
        self.scaler_max = dataset.max(axis=0)
        return self

    def normalize(self, x: np.ndarray) -> np.ndarray:
        """Min-max normalization to [feature_range]."""
        lo, hi = self.feature_range
        rng = self.scaler_max - self.scaler_min
        rng[rng == 0] = 1  # avoid div-by-zero
        return lo + (x - self.scaler_min) / rng * (hi - lo)

    def aleatory_uncertainty(self, x: np.ndarray) -> float:
        """
        Eq. 9:  Ua = sqrt( (1/N) * Σ(xi - μ)² )
        Measures statistical variance of the input sample.
        """
        mu = x.mean()
        return float(np.sqrt(np.mean((x - mu) ** 2)))

    def epistemic_uncertainty(
            self, x_estimated: np.ndarray, x_actual: np.ndarray) -> float:
        """
        Eq. 10: Ue = max(|x̂i - xi|) / √N
        Measures systematic deviation between estimated and actual values.
        """
        N = len(x_actual)
        return float(np.max(np.abs(x_estimated - x_actual)) / np.sqrt(N))

    def compute_delta(
            self, sigma_i: float, sigma_e: float) -> float:
        """
        Eq. 5:  δ = √(σi² + σe²)
        Combined uncertainty parameter used in membership bounds.
        """
        return float(np.sqrt(sigma_i**2 + sigma_e**2))

    def process(
            self,
            raw: np.ndarray,
            estimated: Optional[np.ndarray] = None) -> InputBundle:
        """Full pipeline: normalize → quantify uncertainty → return bundle."""
        normalized = self.normalize(raw)

        # Aleatory (Eq. 9)
        Ua = self.aleatory_uncertainty(normalized)

        # Epistemic (Eq. 10) – use self-estimate if not provided
        x_est = estimated if estimated is not None else normalized * 1.02
        Ue = self.epistemic_uncertainty(x_est, normalized)

        # Total uncertainty (Eq. 8)
        U_total = Ua + Ue

        # δ parameter (Eq. 5)
        delta = self.compute_delta(self.sigma_inherent, self.sigma_env)

        return InputBundle(
            raw_values  = raw,
            normalized  = normalized,
            aleatory_u  = Ua,
            epistemic_u = Ue,
            total_u     = U_total,
            delta       = delta
        )

##############################################################
#  DRFIS – Layer 2: Fuzzy Engine (Inference Layer)          #
##############################################################

import numpy as np
from typing import List, Dict, Callable, Tuple

# ─── Uncertainty-Aware Membership Functions ─────────────────

class UncertaintyAwareMF:
    """
    Gaussian membership function with uncertainty bounds.

    Lower bound  μL(x) = exp( -(x-d)² / 2(σ+δ)² )   Eq. 3
    Upper bound  μU(x) = exp( -(x-d)² / 2(σ-δ)² )   Eq. 4

    d : center parameter
    σ : standard deviation (width)
    δ : uncertainty parameter from InputProcessingLayer
    """

    def __init__(self, center: float, sigma: float,
                 learning_rate: float = 0.01):
        self.d  = center
        self.sigma = sigma
        self.eta   = learning_rate

    def lower_bound(self, x: float, delta: float) -> float:
        """Eq. 3 – lower membership (wider Gaussian = more uncertain)."""
        denom = 2 * (self.sigma + delta) ** 2
        if denom <= 0: return 0.0
        return float(np.exp(-((x - self.d)**2) / denom))

    def upper_bound(self, x: float, delta: float) -> float:
        """Eq. 4 – upper membership (narrower Gaussian = less uncertain)."""
        sigma_eff = max(self.sigma - delta, 1e-6)
        denom = 2 * sigma_eff ** 2
        return float(np.exp(-((x - self.d)**2) / denom))

    def crisp(self, x: float, delta: float) -> float:
        """Midpoint of [lower, upper] used as crisp activation."""
        return (self.lower_bound(x, delta) + self.upper_bound(x, delta)) / 2

    def adapt(self,
              grad_center: float,
              grad_sigma : float) -> None:
        """
        Eq. 6 & 7 – Adaptive Parameter Adjustment
          d(t+1) = d(t) + η∇c(t)
          σ(t+1) = σ(t) + η∇σ(t)
        """
        self.d     += self.eta * grad_center
        self.sigma  = max(0.01, self.sigma + self.eta * grad_sigma)

# ─── Fuzzy Rule ─────────────────────────────────────────────

@dataclass
class FuzzyRule:
    antecedents : List[Tuple[int, str]]  # [(feature_idx, label), ...]
    consequent  : float                   # crisp output value
    weight      : float = 1.0

# ─── Fuzzy Engine ────────────────────────────────────────────

class FuzzyEngine:
    """
    Dynamic rule-based fuzzy inference engine.
    Uses Mamdani-style min-activation with centroid defuzzification.
    Rule weights are adapted based on confidence level in real-time.
    """

    def __init__(self):
        self.mf_library : Dict[str, UncertaintyAwareMF] = {}
        self.rules      : List[FuzzyRule] = []

    def add_mf(self, label: str, mf: UncertaintyAwareMF) -> None:
        self.mf_library[label] = mf

    def add_rule(self, rule: FuzzyRule) -> None:
        self.rules.append(rule)

    def _activate_rule(
            self,
            rule: FuzzyRule,
            x: np.ndarray,
            delta: float) -> float:
        """Min-AND aggregation of antecedent activations."""
        activations = []
        for feat_idx, label in rule.antecedents:
            mf = self.mf_library.get(label)
            if mf is None:
                activations.append(0.0)
                continue
            activations.append(mf.crisp(x[feat_idx], delta))
        return min(activations) if activations else 0.0

    def infer(
            self,
            x: np.ndarray,
            delta: float,
            confidence: float = 1.0) -> float:
        """
        Weighted centroid defuzzification.
        rule weights updated by confidence level.
        """
        weighted_sum = 0.0
        weight_total = 0.0

        for rule in self.rules:
            activation = self._activate_rule(rule, x, delta)
            effective_w = rule.weight * confidence * activation
            weighted_sum += effective_w * rule.consequent
            weight_total += effective_w

        if weight_total == 0:
            return 0.5  # neutral output
        return float(weighted_sum / weight_total)

    def update_rule_weights(
            self,
            confidence_map: Dict[int, float]) -> None:
        """Real-time rule weight update based on confidence level."""
        for i, rule in enumerate(self.rules):
            if i in confidence_map:
                rule.weight = float(np.clip(confidence_map[i], 0.1, 2.0))

##############################################################
#  DRFIS – Layer 3: Robustness Check Layer                  #
##############################################################

import numpy as np

class RobustnessCheckLayer:
    """
    Monitors system stability and detects perturbations.
    Key functions:
      - sensitivity_analysis()  : dOutput/dInput numerical gradient
      - uncertainty_propagation(): Eq. 11 – Uout = √Σ(∂f/∂xi)²·Uin²
      - compute_robustness_score(): composite stability metric
      - adaptive_threshold()    : dynamic pass/fail threshold
    """

    def __init__(
            self,
            base_threshold: float = 0.85,
            history_window : int   = 20):
        self.base_threshold  = base_threshold
        self.history_window  = history_window
        self.output_history  : list = []
        self.sensitivity_history : list = []

    # ── Sensitivity Analysis ────────────────────────────────────
    def sensitivity_analysis(
            self,
            infer_fn,
            x       : np.ndarray,
            delta   : float,
            epsilon : float = 1e-3) -> np.ndarray:
        """
        Numerical Jacobian: ∂output/∂xi  for each feature i.
        Uses central finite-difference: (f(x+ε) - f(x-ε)) / 2ε
        """
        base_out = infer_fn(x, delta)
        grads = np.zeros(len(x))

        for i in range(len(x)):
            x_plus  = x.copy(); x_plus[i]  += epsilon
            x_minus = x.copy(); x_minus[i] -= epsilon
            grads[i] = (infer_fn(x_plus, delta) -
                        infer_fn(x_minus, delta)) / (2 * epsilon)
        return grads

    # ── Uncertainty Propagation (Eq. 11) ────────────────────────
    def uncertainty_propagation(
            self,
            sensitivity_coeffs : np.ndarray,   # ∂f/∂xi
            input_uncertainties: np.ndarray    # Uin_i
    ) -> float:
        """
        Eq. 11:  Uout = √( Σ (∂f/∂xi)² · Uin_i² )
        Propagates input uncertainty through the inference function.
        """
        return float(np.sqrt(
            np.sum((sensitivity_coeffs ** 2) * (input_uncertainties ** 2))
        ))

    # ── Robustness Score ────────────────────────────────────────
    def compute_robustness_score(
            self,
            output         : float,
            Uout           : float,
            sensitivity_mag: float) -> float:
        """
        Composite robustness score ∈ [0, 1].
        Higher score = more stable and reliable decision.

        score = (1 - Uout_norm) * (1 - sensitivity_norm) * stability_bonus
        """
        self.output_history.append(output)
        if len(self.output_history) > self.history_window:
            self.output_history.pop(0)

        # Temporal stability: low std dev over recent outputs = stable
        hist_arr = np.array(self.output_history)
        stability = 1.0 - min(1.0, np.std(hist_arr) * 5)

        # Normalize metrics (clip to [0,1])
        Uout_norm  = min(1.0, Uout)
        sens_norm  = min(1.0, sensitivity_mag / 10.0)

        score = (1 - Uout_norm) * 0.4 + \
                (1 - sens_norm)  * 0.3 + \
                stability         * 0.3
        return float(np.clip(score, 0.0, 1.0))

    # ── Adaptive Threshold ──────────────────────────────────────
    def adaptive_threshold(self) -> float:
        """
        Dynamically adjusts pass threshold based on recent robustness history.
        In noisy periods, threshold is relaxed; in stable periods, tightened.
        """
        if not self.sensitivity_history:
            return self.base_threshold
        avg_sens = np.mean(self.sensitivity_history[-self.history_window:])
        adjustment = float(np.clip(avg_sens * 0.05, -0.1, 0.1))
        return float(np.clip(self.base_threshold - adjustment, 0.5, 0.99))

    def check(
            self,
            robustness_score: float) -> Tuple[bool, float]:
        """
        Returns (PASS, threshold).
        PASS → proceed to Output Layer.
        FAIL → route to Adaptive Tuning module.
        """
        threshold = self.adaptive_threshold()
        passed    = robustness_score >= threshold
        return passed, threshold

Defuzzification, confidence scoring, output validation, and final decision production.

##############################################################
#  DRFIS – Layer 4: Output Generation Layer                 #
##############################################################

import numpy as np
from dataclasses import dataclass

@dataclass
class Decision:
    crisp_output     : float
    confidence       : float
    robustness_score : float
    uncertainty_out  : float
    label            : str
    valid            : bool

class OutputGenerationLayer:
    """
    Final DRFIS layer.
    1. Defuzzification (centroid / weighted average)
    2. Confidence scoring from robustness index + uncertainty
    3. Output validation against min confidence threshold
    4. Semantic labelling of decisions
    """

    def __init__(
            self,
            min_confidence: float = 0.6,
            labels        : List[str] = None):
        self.min_confidence = min_confidence
        self.labels = labels or [
            "Very Low", "Low", "Medium", "High", "Very High"
        ]

    def confidence_score(
            self,
            robustness  : float,
            U_out       : float) -> float:
        """
        Confidence = robustness × (1 – normalized_Uout)
        Penalises outputs with high propagated uncertainty.
        """
        penalty = min(1.0, U_out * 2)
        return float(np.clip(robustness * (1 - penalty), 0.0, 1.0))

    def label_output(self, crisp: float) -> str:
        """Map crisp output in [0,1] to semantic label."""
        n = len(self.labels)
        idx = int(np.floor(crisp * n))
        idx = min(idx, n - 1)
        return self.labels[idx]

    def validate(
            self,
            crisp     : float,
            confidence: float) -> bool:
        """
        Validation criteria:
          ✔ output in [0, 1]
          ✔ confidence ≥ min_confidence
        """
        return (0.0 <= crisp <= 1.0) and (confidence >= self.min_confidence)

    def generate(
            self,
            raw_output       : float,
            robustness_score : float,
            U_out            : float) -> Decision:
        """Full output pipeline → returns Decision dataclass."""
        crisp      = float(np.clip(raw_output, 0.0, 1.0))
        confidence = self.confidence_score(robustness_score, U_out)
        label      = self.label_output(crisp)
        valid      = self.validate(crisp, confidence)

        return Decision(
            crisp_output     = crisp,
            confidence       = confidence,
            robustness_score = robustness_score,
            uncertainty_out  = U_out,
            label            = label,
            valid            = valid
        )

#  DRFIS – Uncertainty Quantification Standalone Module     #

import numpy as np

class UncertaintyQuantifier:
    """
    Standalone module to compute, track and report all
    uncertainty metrics used in the DRFIS pipeline.
    Covers both aleatory (random) and epistemic (systematic) forms.
    """

    def __init__(self):
        self.history = {"Ua": [], "Ue": [], "Utotal": [], "delta": [], "Uout": []}

    # ── Aleatory (Eq. 9) ────────────────────────────────────────
    def aleatory(self, samples: np.ndarray) -> float:
        """Statistical variance component of uncertainty."""
        Ua = float(np.sqrt(np.mean((samples - samples.mean())**2)))
        self.history["Ua"].append(Ua)
        return Ua

    # ── Epistemic (Eq. 10) ──────────────────────────────────────
    def epistemic(
            self,
            x_hat: np.ndarray,   # estimated
            x    : np.ndarray    # actual
    ) -> float:
        """Systematic deviation between model estimates and truth."""
        Ue = float(np.max(np.abs(x_hat - x)) / np.sqrt(len(x)))
        self.history["Ue"].append(Ue)
        return Ue

    # ── Total (Eq. 8) ────────────────────────────────────────────
    def total(self, Ua: float, Ue: float) -> float:
        Ut = Ua + Ue
        self.history["Utotal"].append(Ut)
        return Ut

    # ── Delta (Eq. 5) ─────────────────────────────────────────────
    def delta(self, sigma_i: float, sigma_e: float) -> float:
        d = float(np.sqrt(sigma_i**2 + sigma_e**2))
        self.history["delta"].append(d)
        return d

    # ── Output Propagation (Eq. 11) ──────────────────────────────
    def output_propagation(
            self,
            sensitivities   : np.ndarray,   # ∂f/∂xi
            input_uncertainties: np.ndarray  # Uin_i
    ) -> float:
        Uout = float(np.sqrt(np.sum(
            (sensitivities**2) * (input_uncertainties**2)
        )))
        self.history["Uout"].append(Uout)
        return Uout

    def report(self) -> Dict:
        """Summary statistics for each tracked metric."""
        rep = {}
        for key, vals in self.history.items():
            if vals:
                rep[key] = {
                    "mean": float(np.mean(vals)),
                    "std" : float(np.std(vals)),
                    "min" : float(np.min(vals)),
                    "max" : float(np.max(vals)),
                }
        return rep

##############################################################
#  DRFIS – Adaptive Tuning Module                           #
#  Implements Eq.6 & Eq.7 (gradient-based parameter update) #
##############################################################

import numpy as np

class AdaptiveTuner:
    """
    Called when Robustness Check FAILS.
    Adjusts MF centers and widths using gradient descent.

    Eq. 6:  d(t+1)  = d(t)  + η·∇c(t)
    Eq. 7:  σ(t+1)  = σ(t)  + η·∇σ(t)

    Gradient estimation via finite differences on inference output.
    """

    def __init__(
            self,
            learning_rate  : float = 0.01,
            momentum       : float = 0.9,
            max_iterations : int   = 50,
            tol            : float = 1e-4):
        self.eta            = learning_rate
        self.momentum       = momentum
        self.max_iterations = max_iterations
        self.tol            = tol
        self._velocity_c    : Dict[str, float] = {}
        self._velocity_s    : Dict[str, float] = {}

    def _estimate_grad_center(
            self, mf, x, delta, engine, epsilon=1e-3) -> float:
        """∂output/∂d via finite difference."""
        orig = mf.d
        mf.d = orig + epsilon
        f_plus  = engine.infer(x, delta)
        mf.d = orig - epsilon
        f_minus = engine.infer(x, delta)
        mf.d = orig
        return (f_plus - f_minus) / (2 * epsilon)

    def _estimate_grad_sigma(
            self, mf, x, delta, engine, epsilon=1e-3) -> float:
        """∂output/∂σ via finite difference."""
        orig = mf.sigma
        mf.sigma = max(0.01, orig + epsilon)
        f_plus   = engine.infer(x, delta)
        mf.sigma = max(0.01, orig - epsilon)
        f_minus  = engine.infer(x, delta)
        mf.sigma = orig
        return (f_plus - f_minus) / (2 * epsilon)

    def tune(
            self,
            engine      : "FuzzyEngine",
            x           : np.ndarray,
            delta       : float,
            target_output: float) -> int:
        """
        Run adaptive gradient-descent tuning.
        Returns number of iterations taken.
        """
        for iteration in range(self.max_iterations):
            current_out = engine.infer(x, delta)
            error = target_output - current_out

            if abs(error) < self.tol:
                break

            for label, mf in engine.mf_library.items():
                grad_c = self._estimate_grad_center(mf, x, delta, engine)
                grad_s = self._estimate_grad_sigma (mf, x, delta, engine)

                # Momentum update (Eq. 6 & 7 with momentum)
                vc = self._velocity_c.get(label, 0.0)
                vs = self._velocity_s.get(label, 0.0)

                vc = self.momentum * vc + self.eta * grad_c * error
                vs = self.momentum * vs + self.eta * grad_s * error

                self._velocity_c[label] = vc
                self._velocity_s[label] = vs

                mf.adapt(vc, vs)  # applies Eq. 6 & 7

        return iteration + 1

class InputProcessingLayer:

    def fit(self, X: np.ndarray):
        self.min_ = X.min(axis=0)
        self.max_ = X.max(axis=0)
        return self

    def normalize(self, x):
        rng = self.max_ - self.min_
        rng[rng==0] = 1
        return (x - self.min_) / rng

    def process(self, raw, estimated=None) -> InputBundle:
        norm = self.normalize(raw)
        mu   = norm.mean()
        Ua   = float(np.sqrt(np.mean((norm - mu)**2)))                    # Eq.9
        x_est = estimated if estimated is not None else norm * 1.02
        Ue    = float(np.max(np.abs(x_est - norm)) / np.sqrt(len(norm)))    # Eq.10
        Ut    = Ua + Ue                                                      # Eq.8
        delta = float(np.sqrt(self.sigma_i**2 + self.sigma_e**2))        # Eq.5
        return InputBundle(raw, norm, Ua, Ue, Ut, delta)

# ═══════════════════════════════════════════════════════════
#  LAYER 2 – FUZZY ENGINE
# ═══════════════════════════════════════════════════════════

class MembershipFunction:
    def __init__(self, center, sigma, eta=0.01):
        self.d, self.sigma, self.eta = center, sigma, eta

    def lower(self, x, delta) -> float:                    # Eq.3
        denom = 2 * (self.sigma + delta)**2
        return float(np.exp(-((x-self.d)**2)/max(denom,1e-9)))

    def upper(self, x, delta) -> float:                    # Eq.4
        s = max(self.sigma - delta, 1e-6)
        return float(np.exp(-((x-self.d)**2)/(2*s**2)))

    def crisp(self, x, delta) -> float:
        return (self.lower(x,delta) + self.upper(x,delta)) / 2

    def adapt(self, gc, gs):                               # Eq.6 & 7
        self.d      += self.eta * gc
        self.sigma   = max(0.01, self.sigma + self.eta * gs)

class FuzzyEngine:
    def __init__(self):
        self.mfs   : Dict[str, MembershipFunction] = {}
        self.rules : List[FuzzyRule] = []

    def add_mf  (self, lbl, mf ): self.mfs[lbl] = mf
    def add_rule(self, rule    ): self.rules.append(rule)

    def infer(self, x, delta, confidence=1.0) -> float:
        ws, wt = 0.0, 0.0
        for r in self.rules:
            acts = [self.mfs[lbl].crisp(x[i],delta)
                    for i, lbl in r.antecedents
                    if lbl in self.mfs]
            a = min(acts) if acts else 0.0
            ew = r.weight * confidence * a
            ws += ew * r.consequent
            wt += ew
        return ws/wt if wt > 0 else 0.5

# ═══════════════════════════════════════════════════════════
#  LAYER 3 – ROBUSTNESS CHECK
# ═══════════════════════════════════════════════════════════

class RobustnessCheckLayer:
    def __init__(self, base_thresh=0.75, window=20):
        self.thresh  = base_thresh
        self.history = []

    def sensitivity(self, fn, x, delta, eps=1e-3):
        g = np.zeros(len(x))
        for i in range(len(x)):
            xp, xm = x.copy(), x.copy()
            xp[i]+=eps; xm[i]-=eps
            g[i] = (fn(xp,delta) - fn(xm,delta)) / (2*eps)
        return g

    def propagate(self, sens, Uin):                        # Eq.11
        return float(np.sqrt(np.sum((sens**2)*(Uin**2))))

    def score(self, output, Uout, sens_mag) -> float:
        self.history.append(output)
        if len(self.history) > 20: self.history.pop(0)
        stab = 1.0 - min(1.0, np.std(self.history)*5)
        s = (1-min(1,Uout))*0.4 + (1-min(1,sens_mag/10))*0.3 + stab*0.3
        return float(np.clip(s,0,1))

    def check(self, rob_score) -> Tuple[bool, float]:
        return rob_score >= self.thresh, self.thresh

# ═══════════════════════════════════════════════════════════
#  LAYER 4 – OUTPUT GENERATION
# ═══════════════════════════════════════════════════════════

LABELS = ["Very Low", "Low", "Medium", "High", "Very High"]

class OutputGenerationLayer:
    def __init__(self, min_conf=0.5):
        self.min_conf = min_conf

    def generate(self, raw, rob, Uout) -> Decision:
        crisp = float(np.clip(raw, 0, 1))
        conf  = float(np.clip(rob * (1 - min(1,Uout*2)), 0, 1))
        lbl   = LABELS[min(int(crisp*5), 4)]
        valid = (0<=crisp<=1) and (conf>=self.min_conf)
        return Decision(crisp, conf, rob, Uout, lbl, valid)

# ═══════════════════════════════════════════════════════════
#  DRFIS ORCHESTRATOR
# ═══════════════════════════════════════════════════════════

class DRFIS:
    """
    Full 4-layer DRFIS pipeline orchestrator.
    Usage:
        drfis = DRFIS.build_default()
        decision = drfis.run(np.array([0.6, 0.4]))
        print(decision)
    """

    def __init__(self, ipl, engine, rcl, ogl):
        self.ipl    = ipl
        self.engine = engine
        self.rcl    = rcl
        self.ogl    = ogl
        self.run_log = []

    @classmethod
    def build_default(cls) -> "DRFIS":
        """Build a default DRFIS with sample rules for 2-feature input."""
        # --- Layer 1
        dummy = np.random.rand(100, 2)
        ipl = InputProcessingLayer(sigma_i=0.10, sigma_e=0.05)
        ipl.fit(dummy)

        # --- Layer 2
        engine = FuzzyEngine()
        for label, center in [("low",0.2),("medium",0.5),("high",0.8)]:
            engine.add_mf(label, MembershipFunction(center, 0.15))

        engine.add_rule(FuzzyRule([(0,"low"),   (1,"low")],    0.1))
        engine.add_rule(FuzzyRule([(0,"low"),   (1,"medium")], 0.3))
        engine.add_rule(FuzzyRule([(0,"medium"),(1,"medium")], 0.5))
        engine.add_rule(FuzzyRule([(0,"medium"),(1,"high")],   0.7))
        engine.add_rule(FuzzyRule([(0,"high"),  (1,"high")],   0.9))

        return cls(ipl, engine, RobustnessCheckLayer(), OutputGenerationLayer())

    def run(self, raw: np.ndarray) -> Decision:
        # L1: Input Processing
        bundle = self.ipl.process(raw)

        # L2: Fuzzy Inference
        raw_out = self.engine.infer(bundle.normalized, bundle.delta)

        # L3: Robustness Check
        sens     = self.rcl.sensitivity(lambda x,d: self.engine.infer(x,d),
                                        bundle.normalized, bundle.delta)
        Uout     = self.rcl.propagate(sens, np.full(len(raw), bundle.total_u))
        rob      = self.rcl.score(raw_out, Uout, float(np.linalg.norm(sens)))
        passed, _ = self.rcl.check(rob)

        # If robustness fails → log warning (production would trigger AdaptiveTuner)
        if not passed:
            print(f"[WARN] Robustness check FAILED (score={rob:.3f}). Tuning needed.")

        # L4: Output
        decision = self.ogl.generate(raw_out, rob, Uout)
        self.run_log.append(decision)
        return decision
